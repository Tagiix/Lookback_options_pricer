92SpotResults
93MonteCarloSimulator::simulateSinglePath(double initialSpot, double rate,
94 double volatility,
95 double timeToMaturity) const {
96
97 SpotResults results;
98
99 double dt = timeToMaturity / numTimeSteps_;
100 double S = initialSpot;
101 double S_max = S;
102 double S_min = S;
103
104 double drift = (rate - 0.5 * volatility * volatility) * dt;
105 double sigma = volatility * std::sqrt(dt);
106
107 for (unsigned long i = 0; i < numTimeSteps_; ++i) {
108 double dW = generateStandardNormal();
109 S *= std::exp(drift + sigma * dW);
110 S_max = std::max(S_max, S);
111 S_min = std::min(S_min, S);
112 }
113
114 // Broadie-Glasserman-Kou continuity correction: adjust the discrete
115 // extrema to approximate continuous monitoring.
116 if (bgkCorrection_) {
117 double correction = std::exp(BGK_BETA1 * volatility * std::sqrt(dt));
118 S_min /= correction; // continuous min is lower than discrete min
119 S_max *= correction; // continuous max is higher than discrete max
120 }
121
122 results.spotAtMaturity = S;
123 results.maximumSpot = S_max;
124 results.minimumSpot = S_min;
125
126 return results;
127}
