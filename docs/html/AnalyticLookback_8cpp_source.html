30double LookbackAnalyticSolution::delta_plus(double tau, double s, double r,
31 double sigma) {
32 assert(s > 0);
33 return 1.0 / (sigma * std::sqrt(tau)) *
34 (std::log(s) + (r + 0.5 * sigma * sigma) * tau);
35}

37double LookbackAnalyticSolution::delta_minus(double tau, double s, double r,
38 double sigma) {
39 assert(s > 0);
40 return 1.0 / (sigma * std::sqrt(tau)) *
41 (std::log(s) + (r - 0.5 * sigma * sigma) * tau);
42}

44double LookbackAnalyticSolution::price(OptionType option, double S0, double T,
45 double sigma, double r) {
46 assert(S0 > 0);
47 assert(T > 0);
48 assert(sigma > 0);
49 double delta_p = delta_plus(T, 1.0, r, sigma);
50 double delta_m = delta_minus(T, 1.0, r, sigma);
51 if (option == OptionType::Call) {
52 double V_discounted =
53 S0 * normalCDF(delta_p) - S0 * std::exp(-r * T) * normalCDF(delta_m) +
54 std::exp(-r * T) * (sigma * sigma * S0 / (2.0 * r)) *
55 normalCDF(delta_m) -
56 normalCDF(-delta_p) * S0 * (sigma * sigma / (2.0 * r));
57 return V_discounted;
58 } else { // Put
59 return S0 * std::exp(-r * T) * normalCDF(-delta_m) +
60 S0 * normalCDF(delta_p) * (1 + (sigma * sigma) / (2 * r)) -
61 std::exp(-r * T) * S0 * (sigma * sigma / (2.0 * r)) *
62 normalCDF(-delta_m) -
63 S0;
64 }
65}

67double LookbackAnalyticSolution::delta(OptionType option, double S0, double T,

74double LookbackAnalyticSolution::gamma(OptionType option, double S0, double T,
75 double sigma, double r) {
