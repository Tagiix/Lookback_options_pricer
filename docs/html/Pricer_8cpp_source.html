69static double computeStdError(const std::vector<double> &values) {
70 size_t N = values.size();
71 double mean = std::accumulate(values.begin(), values.end(), 0.0) / N;
72 double variance = std::accumulate(values.begin(), values.end(), 0.0,
73 [mean](double sum, double val) {
74 return sum + (val - mean) * (val - mean);
75 }) /
76 (N - 1);
77 return std::sqrt(variance / N);
78}

80// Stderror is passed by reference to return both price and its standard error
81// from the same function.
82// If storeValues is true, the individual discounted payoffs are stored in
83// values, which allows computing standard errors for Greeks via finite
84// differences on per-simulation values.
85double PricerLookbackOption::computeRawPrice(double spot, double rate,
86 double volatility, double maturity,
87 double &stdError,
88 unsigned int seed,
89 std::vector<double> &values,
90 bool storeValues) {
91 simulator_.setRandomSeed(seed);
92 // Simulate paths
93 auto paths = simulator_.simulatePaths(spot, rate, volatility, maturity);
94 // Compute payoffs
95 std::vector<double> payoffs(paths.size());
96 for (size_t i = 0; i < paths.size(); ++i) {
97 if (option_.getType() == OptionType::Call) {
98 payoffs[i] =
99 option_.payoff(paths[i].minimumSpot, paths[i].spotAtMaturity);
100 } else {
101 payoffs[i] =
102 option_.payoff(paths[i].maximumSpot, paths[i].spotAtMaturity);
103 }
104 }
105 // Discount payoffs and compute price
106 double discountFactor = std::exp(-rate * maturity);
107 double price = discountFactor *
108 std::accumulate(payoffs.begin(), payoffs.end(), 0.0) /
109 payoffs.size();
110 // Compute standard error
111 double payoffMean =
112 std::accumulate(payoffs.begin(), payoffs.end(), 0.0) / payoffs.size();
113 double payoffVariance =
114 std::accumulate(payoffs.begin(), payoffs.end(), 0.0,
115 [payoffMean](double sum, double payoff) {
116 return sum +
117 (payoff - payoffMean) * (payoff - payoffMean);
118 }) /
119 (payoffs.size() - 1);
120 stdError = discountFactor * std::sqrt(payoffVariance / payoffs.size());
121
122 // Store individual discounted payoffs if requested
123 if (storeValues) {
124 values.resize(payoffs.size());
125 for (size_t i = 0; i < payoffs.size(); ++i) {
126 values[i] = discountFactor * payoffs[i];
127 }
128 }
129
130 return price;
131}

133PricingResult PricerLookbackOption::compute(bool withGreekStdErrors) {
134 PricingResult result;
135 // needed to use the same randomness for all price computations to ensure that
136 // the differences are due to parameter changes and not random noise
137 unsigned int seed = simulator_.getRandomSeed();
138 double stdError;
139
140 // Vectors to store per-simulation discounted payoffs for Greek std errors.
141 // Only populated when withGreekStdErrors is true.
142 std::vector<double> baseValues, upSpotValues, downSpotValues;
143 std::vector<double> forwardTimeValues, backwardTimeValues;
144 std::vector<double> upRateValues, downRateValues;
145 std::vector<double> upVolValues, downVolValues;
146 // Unused placeholder for calls that don't need stored values
147 std::vector<double> unused;
148
149 // Compute price and standard error for the base parameters
150 // (price std error is always computed via the stdError out-parameter)
151 result.price = computeRawPrice(
152 MarketParameters_.getSpot(), MarketParameters_.gtetRiskFreeRate(),
153 MarketParameters_.getVolatility(), option_.getMaturity(), stdError, seed,
154 withGreekStdErrors ? baseValues : unused, withGreekStdErrors);
155 result.priceStd = stdError;
156
157 // Compute Greeks using finite differences
158
159 // Delta = dV/dS ~ (V(S+ds) - V(S-ds)) / (2*ds)
160 double spotBumpAbs = MarketParameters_.getSpot() * SPOT_BUMP;
161 double priceSpotUp = computeRawPrice(
162 MarketParameters_.getSpot() * (1 + SPOT_BUMP),
163 MarketParameters_.gtetRiskFreeRate(), MarketParameters_.getVolatility(),
164 option_.getMaturity(), stdError, seed,
165 withGreekStdErrors ? upSpotValues : unused, withGreekStdErrors);
166
167 double priceSpotDown = computeRawPrice(
168 MarketParameters_.getSpot() * (1 - SPOT_BUMP),
169 MarketParameters_.gtetRiskFreeRate(), MarketParameters_.getVolatility(),
170 option_.getMaturity(), stdError, seed,
171 withGreekStdErrors ? downSpotValues : unused, withGreekStdErrors);
172
173 result.delta = (priceSpotUp - priceSpotDown) / (2 * spotBumpAbs);
174
175 // Gamma = d2V/dS2 ~ (V(S+ds) - 2*V(S) + V(S-ds)) / (ds^2)
176 double spotBumpAbs2 = spotBumpAbs * spotBumpAbs;
177 result.gamma =
178 (priceSpotUp - 2 * result.price + priceSpotDown) / spotBumpAbs2;
179
180 if (withGreekStdErrors) {
181 size_t N = baseValues.size();
182 std::vector<double> greekPerSim(N);
183
184 // Delta std error
185 for (size_t i = 0; i < N; ++i) {
186 greekPerSim[i] =
187 (upSpotValues[i] - downSpotValues[i]) / (2 * spotBumpAbs);
188 }
189 result.deltaStd = computeStdError(greekPerSim);
190
191 // Gamma std error
192 for (size_t i = 0; i < N; ++i) {
193 greekPerSim[i] =
194 (upSpotValues[i] - 2 * baseValues[i] + downSpotValues[i]) /
195 spotBumpAbs2;
196 }
197 result.gammaStd = computeStdError(greekPerSim);
198 } else {
199 result.deltaStd = 0.0;
200 result.gammaStd = 0.0;
201 }
202
203 // Theta = dV/dT ~ (V(T+dt) - V(T-dt)) / (2*dt)
204 double priceAfter = computeRawPrice(
205 MarketParameters_.getSpot(), MarketParameters_.gtetRiskFreeRate(),
206 MarketParameters_.getVolatility(),
207 option_.getMaturity() * (1.0 + TIME_BUMP), stdError, seed,
208 withGreekStdErrors ? forwardTimeValues : unused, withGreekStdErrors);
209
210 double priceBefore = computeRawPrice(
211 MarketParameters_.getSpot(), MarketParameters_.gtetRiskFreeRate(),
212 MarketParameters_.getVolatility(),
213 option_.getMaturity() * (1.0 - TIME_BUMP), stdError, seed,
214 withGreekStdErrors ? backwardTimeValues : unused, withGreekStdErrors);
215
216 result.theta =
217 (priceAfter - priceBefore) / (2 * option_.getMaturity() * TIME_BUMP);
218
219 // Rho = dV/dr ~ (V(r+dr) - V(r-dr)) / (2*dr)
220 double priceRateUp = computeRawPrice(
221 MarketParameters_.getSpot(),
222 MarketParameters_.gtetRiskFreeRate() * (1.0 + RATE_BUMP),
223 MarketParameters_.getVolatility(), option_.getMaturity(), stdError, seed,
224 withGreekStdErrors ? upRateValues : unused, withGreekStdErrors);
225
226 double priceRateDown = computeRawPrice(
227 MarketParameters_.getSpot(),
228 MarketParameters_.gtetRiskFreeRate() * (1.0 - RATE_BUMP),
229 MarketParameters_.getVolatility(), option_.getMaturity(), stdError, seed,
230 withGreekStdErrors ? downRateValues : unused, withGreekStdErrors);
231
232 result.rho = (priceRateUp - priceRateDown) /
233 (2 * RATE_BUMP * MarketParameters_.gtetRiskFreeRate());
234
235 // Vega = dV/dsigma ~ (V(sigma+dv) - V(sigma-dv)) / (2*dv)
236 double volBumpAbs = MarketParameters_.getVolatility() * VOL_BUMP;
237 double priceVolUp = computeRawPrice(
238 MarketParameters_.getSpot(), MarketParameters_.gtetRiskFreeRate(),
239 MarketParameters_.getVolatility() * (1.0 + VOL_BUMP),
240 option_.getMaturity(), stdError, seed,
241 withGreekStdErrors ? upVolValues : unused, withGreekStdErrors);
242
243 double priceVolDown = computeRawPrice(
244 MarketParameters_.getSpot(), MarketParameters_.gtetRiskFreeRate(),
245 MarketParameters_.getVolatility() * (1.0 - VOL_BUMP),
246 option_.getMaturity(), stdError, seed,
247 withGreekStdErrors ? downVolValues : unused, withGreekStdErrors);
248
249 result.vega = (priceVolUp - priceVolDown) / (2 * volBumpAbs);
250
251 if (withGreekStdErrors) {
252 size_t N = baseValues.size();
253 std::vector<double> greekPerSim(N);
254
255 // Theta std error
256 for (size_t i = 0; i < N; ++i) {
257 greekPerSim[i] = (forwardTimeValues[i] - backwardTimeValues[i]) /
258 (2 * option_.getMaturity() * TIME_BUMP);
259 }
260 result.thetaStd = computeStdError(greekPerSim);
261
262 // Rho std error
263 for (size_t i = 0; i < N; ++i) {
264 greekPerSim[i] = (upRateValues[i] - downRateValues[i]) / (2 * RATE_BUMP);
265 }
266 result.rhoStd = computeStdError(greekPerSim);
267
268 // Vega std error
269 for (size_t i = 0; i < N; ++i) {
270 greekPerSim[i] = (upVolValues[i] - downVolValues[i]) / (2 * volBumpAbs);
271 }
272 result.vegaStd = computeStdError(greekPerSim);
273 } else {
274 result.thetaStd = 0.0;
275 result.rhoStd = 0.0;
276 result.vegaStd = 0.0;
277 }
278
279 return result;
280}

282PriceAndDelta PricerLookbackOption::priceAndDeltaForSpot(double spot) {
283 unsigned int seed = simulator_.getRandomSeed();
284 double stdError;
285 std::vector<double> unused;
286 double price =
287 computeRawPrice(spot, MarketParameters_.gtetRiskFreeRate(),
288 MarketParameters_.getVolatility(), option_.getMaturity(),
289 stdError, seed, unused, false);
290 double error = stdError;
291 // Compute Delta using finite difference
292 double priceUp = computeRawPrice(
293 spot * (1 + SPOT_BUMP), MarketParameters_.gtetRiskFreeRate(),
294 MarketParameters_.getVolatility(), option_.getMaturity(), stdError, seed,
295 unused, false);
296 double priceDown = computeRawPrice(
297 spot * (1 - SPOT_BUMP), MarketParameters_.gtetRiskFreeRate(),
298 MarketParameters_.getVolatility(), option_.getMaturity(), stdError, seed,
299 unused, false);
300 double delta = (priceUp - priceDown) / (2 * spot * SPOT_BUMP);
301 return {price, error, delta};
302}
