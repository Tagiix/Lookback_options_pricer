35double LookbackOption::payoff(double extremumOfSpot,
36 double SpotAtMaturity) const {
37 // extremumOfSpot will need to be different for call and put options: for a
38 // call option, it is the minimum
39 if (optionType_ == OptionType::Call) {
40 return std::max(0.0, SpotAtMaturity - extremumOfSpot);
41 } else { // OptionType::Put
42 return std::max(0.0, extremumOfSpot - SpotAtMaturity);
43 }
